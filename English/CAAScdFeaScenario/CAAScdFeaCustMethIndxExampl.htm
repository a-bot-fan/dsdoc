Example Custom Method Index
Technical Article

Abstract

This article presents an example custom Method index file.

 

<?xml version="1.0" encoding="utf-8"?>

<!--
  Example custom Method index file.
  File name must start with the prefix 'SMAMpaMethodIndex'.
-->

<!--
<Index  name=""?  nlsFile=""? >
-->

<Index  name="Index1"   >

    <!--
    <Method  name=""?  description=""?  nlsFile=""?  file="" />
    -->

    <Method file="TestStructural" name="Structural Builtins" description="Structural with builtin Actions" />

    <Method file="TestStructuralCustom" name="Structural Custom" description="Structural with custom restraint and load Actions" />

</Index>
